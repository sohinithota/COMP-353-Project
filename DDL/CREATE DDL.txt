create table patient (
    patientID	      int               NOT NULL,
	patfname          varchar(15)       NOT NULL,
    patminit          char,                  
    patlname          varchar(15)       NOT NULL,
    ssn               char(9)           NOT NULL,
    bdate             DATE,
    address           varchar(30), 
    sex               char, 
    arrivalTime       TIMESTAMP         NOT NULL DEFAULT CURRENT_TIMESTAMP,
	departTime	      TIMESTAMP         DEFAULT CURRENT_TIMESTAMP,
	medcondition	  varchar(40)    		        NOT NULL,
	adminID	          int	         		        NOT NULL,
	inBed			  char,
	bedID	          int, 	
    primary key (patientID)
    -- foreign key (adminID) references patientdocadmin(adminID)
    -- foreign key (bedID) references bed(bedID)
) ;

create table doctor (
    docID int NOT NULL,
	adminID int NOT NULL,
    speciality varchar(15) NOT NULL,
	docfname varchar(15) NOT NULL,
	doclname varchar(15) NOT NULL,
	deviceID int NOT NULL,
    primary key (docID)
    -- foreign key (adminID) references patientdocadmin(adminID),	
	-- foreign key (deviceID) references medicaldevices(deviceID)	
) ;


create table assignment(
    adminID               int               NOT NULL, 
    patientID	          int               NOT NULL,
    docID	          int               NOT NULL,  
    primary key (patientID,docID,adminID)
) ;



create table medicaldevices (
    deviceID              int		    NOT NULL,
    devicename        varchar(15),
    primary key (deviceID)
) ;


create table bed (
    bedID                 int		    NOT NULL,
    occupied         CHAR(1),
    clean            CHAR(1),
    primary key (bedID)
) ;

create table patientAdministrator (
    adminID int NOT NULL,
    adminfname DATE,  
    adminlname varchar(15) NOT NULL,
    primary key (adminID)
) ;
